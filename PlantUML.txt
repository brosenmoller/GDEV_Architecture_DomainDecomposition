@startuml
!theme carbon-gray

class Player
{
  +int Health
  -WeaponManager weaponManager;
  -KeepsakeManager keepsakeManager;
}

struct Damage
{
  +int MainDamage;
  +int BackstabDamage;
  +bool IsPoisned;
}

interface IDamageable
{
  +void TakeDamage(Damage damage);
}

abstract class Weapon
{
  +void ApplyEffect(out Damage damage);
}

class WeaponManager
{
  -Weapon[] unlockedWeapons;
  -Weapon[] equipedWeapons;
}

class KeepsakeManager
{
  -Keepsake[] unlockedKeepsakes;
  -Keepsake[] equipedKeepsakes;
  
  +void ApplyEffect(...);
}

abstract class Keepsake
{
   +void ApplyEffect(...)
}

Player *-- WeaponManager
WeaponManager *-- Weapon

Player *-- KeepsakeManager
KeepsakeManager *-- Keepsake

Weapon .. Damage
IDamageable .. Damage

@enduml
